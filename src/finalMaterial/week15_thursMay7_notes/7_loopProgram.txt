def diff(big: Z, small: Z): Z = {
    Contract(
        Requires( big >= small ),
        Ensures ( Res[Z] == big-small )
    )

    //PREMISES:
    big >= small (precondition, can ALWAYS list as a premise)

    var count: Z = 0
    var cur: Z = big

    //PREMISES:
    big >= small (precondition, can ALWAYS list as a premise)
    count == 0
    cur == big


    //SHOW:
    count == (big-cur) (must prove ALL invariants before the loop begins)


    while (cur != small) {
        Invariant(
            Modifies(count, cur),
            count == (big-cur)
        )

        //PREMISES:
        big >= small (precondition, can ALWAYS list as a premise)
        count == (big-cur) (list ALL invariants as premise at beginning of iteration)
        cur != small (loop condition is true)

        cur = cur - 1
        count = count + 1

        //SHOW:
        count == (big-cur) (must prove ALL invariants still hold at end of iteration)

    }

    //PREMISES:
    big >= small (precondition, can ALWAYS list as a premise)
    count == (big-cur) (can list ALL invariants as premises after loop ends)
    !(cur != small) (loop condition is false)


    //SHOW:
    count == big-small (must prove ALL postconditions)


    return count
}
